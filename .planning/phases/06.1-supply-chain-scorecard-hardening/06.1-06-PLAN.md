    ```yaml
    name: Docs

    # Phase 6.1 / D-37 — MkDocs Material site deployed to GitHub Pages.
    #
    # Triggers:
    #   - push to main → build + deploy to gh-pages
    #   - pull_request → build-only (validate diff)
    #   - workflow_dispatch → manual rebuild (e.g. after a content-only update
    #     that doesn't trigger CI)
    #
    # Permissions: top-level read-all; deploy job escalates to pages+id-token+contents:read.

    on:
      push:
        branches: [main]
      pull_request:
        branches: ["**"]
      workflow_dispatch:

    permissions: read-all

    # Allow only one concurrent deployment; cancel queued runs but let
    # in-progress runs finish (matches GitHub's recommended pages pattern).
    concurrency:
      group: pages
      cancel-in-progress: false

    jobs:
      build:
        name: Build mkdocs site
        runs-on: ubuntu-24.04
        permissions:
          contents: read
        steps:
          - name: Checkout
            uses: actions/checkout@<SHA> # v4.x.x   # reuse SHA pinned in Plan 06.1-01 ci.yml line 23
            with:
              fetch-depth: 0
              persist-credentials: false

          - name: Set up Python 3.12
            uses: actions/setup-python@<SHA> # v5.x.x   # same SHA as sast.yml (Plan 06.1-03)
            with:
              python-version: "3.12"

          - name: Install mkdocs + plugins (pinned)
            run: |
              pip install \
                mkdocs==1.6.1 \
                mkdocs-material==9.5.41 \
                mkdocs-static-i18n==1.2.3 \
                mkdocs-include-markdown-plugin==7.0.0
              # Pinned versions for deterministic builds. Bump via manual PR;
              # Dependabot does not currently scan pip ecosystem (see Plan
              # 06.1-05 backlog note).

          - name: Build site (strict mode)
            run: mkdocs build --strict

          - name: Upload artifact for deploy job
            if: github.event_name == 'push' && github.ref == 'refs/heads/main'
            uses: actions/upload-pages-artifact@<SHA> # v3.x.x   # resolve via gh api repos/actions/upload-pages-artifact/git/ref/tags/<latest>
            with:
              path: site/

      deploy:
        name: Deploy to GitHub Pages
        needs: build
        if: github.event_name == 'push' && github.ref == 'refs/heads/main'
        runs-on: ubuntu-24.04
        permissions:
          pages: write       # required for actions/deploy-pages
          id-token: write    # required for OIDC verification by deploy-pages
          contents: read
        environment:
          name: github-pages
          url: ${{ steps.deployment.outputs.page_url }}
        steps:
          - name: Deploy to GitHub Pages
            id: deployment
            uses: actions/deploy-pages@<SHA> # v4.x.x   # resolve via gh api repos/actions/deploy-pages/git/ref/tags/<latest>
    ```
